namespace = knightly

#Character gets an event when they are made someone's squire
character_event = {
	id = knightly.001
	desc = "161001_desc"
	picture = GFX_evt_child_sword
	is_triggered_only = yes
	option = {
		name = knightly.001.1
		ai_chance = {
			factor = 100
		}
		effect = {
			add_trait = squire
			prestige = 10
		}
	}
}

#Character gets an event when they are personally knighted

		}
		effect = {

#Character gets an event when they become an adult to convert squire into knight
character_event = {
	id = knightly.003
	desc = "161003_desc"

	}

		name = knightly.003.1
		ai_chance = {
			factor = 100
		}
		effect = {
			remove_trait = squire
			add_trait = knight

		}
	}
}
