#Knighted for valor

	id = knightly.004
	desc = "161004_desc"
	picture = GFX_event_knighthood1
	is_triggered_only = yes

	trigger = {
		is_ruler = yes
		is_feudal = yes
		religion_group = christian
		martial = 12
		NOT = { trait = knight }
		NOR = {
			trait = craven
			trait = incapable
			trait = inbred
			trait = infirm
		}
	}

	option = {
		name = knightly.004.1
		ai_chance = {
			factor = 100
		}
		ROOT = { 
			add_trait = knight
			prestige = 100
		}
	}
}

		NOT = {

		}
	}
}
